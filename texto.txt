-- Criação das tabelas & Alterações

create table clientes (
	id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(255) NOT NULL,
    data_de_nascimento DATE,
    sexo VARCHAR(30),
    endereco_id int,
    email VARCHAR(100) UNIQUE NOT NULL,
    senha varchar(100) NOT NULL
);

-- Criação da chave estrangeira na tabela clientes,ligando a tabela de endereço
ALTER TABLE clientes ADD FOREIGN KEY(endereco_id)  REFERENCES endereco(id);

create table endereco (
	id int primary key auto_increment,
    id_cliente int,
	estado varchar (2),
    cidade varchar(75) not null,
    rua varchar(75) not null,
    bairro varchar(75) not null,
    foreign key (id_cliente) references clientes(id)
);

create table livros (
	id_livro int,
    titulo VARCHAR(200) NOT NULL,
    genero VARCHAR(30),
    estoque_id INT
);

-- adicao da chave estrangeira estoque na tabela de livros

ALTER TABLE livros ADD FOREIGN KEY(estoque_id) REFERENCES estoque(id);

create table carrinho (
	id int primary key auto_increment,
    cliente_id int,
    livro_id int,
    quantidade int,
    foreign key(cliente_id) references clientes(id),
    foreign key(livro_id) references livros(id)
);

create table pedidos (
	id INT PRIMARY KEY AUTO_INCREMENT,
    id_cliente INT,
    id_livro INT,
    quantidade INT,
    valor_da_venda DECIMAL(10, 2),
    FOREIGN KEY (id_cliente) REFERENCES clientes(id),
    FOREIGN KEY (id_livro) REFERENCES livros(id)
);

create table estoque (
	id INT PRIMARY KEY AUTO_INCREMENT,
    nome varchar(255),
    quantidade INT,
    valor_de_custo decimal(10, 2),
    descricao varchar(1000)
);
